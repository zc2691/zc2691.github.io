scatterplot

instacart %>% 
  count(aisle) %>% 
  filter(n > 10000) %>%
  mutate(
    text_label = str_c("Aisle: ", aisle, "\nCount: ", n)) %>% 
  mutate(aisle = fct_reorder(aisle, n)) %>% 
  plot_ly(x = ~aisle, y = ~n, type = "scatter", mode = "markers",
          text = ~text_label)

  summarize(order_hour_of_day_mean = mean(order_hour_of_day)) %>%
